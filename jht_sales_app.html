function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName("Sales") || 
    SpreadsheetApp.getActiveSpreadsheet()
    .getActiveSheet();
  
  var data = JSON.parse(e.postData.contents);
  
  // Add header if empty
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Timestamp","Date","Channel",
      "Order Ref","Customer","Contact",
      "Brand","Product","Qty","Price",
      "Line Total","Grand Total","Notes"
    ]);
  }
  
  var items = data.items;
  var grandTotal = data.grandTotal;
  
  items.forEach(function(item, i) {
    sheet.appendRow([
      new Date(),
      data.date,
      data.channel,
      data.orderRef,
      data.customer,
      data.contact,
      item.brand,
      item.product,
      item.qty,
      item.price,
      item.lineTotal,
      i === 0 ? grandTotal : "",
      i === 0 ? data.notes : ""
    ]);
  });
  
  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
